<script>

/* ---------------- LOAD STATUS ---------------- */

async function fetchLoad(){

const response = await fetch("http://localhost:8080/load/status");

const data = await response.json();

document.getElementById("loadValue").innerText = data.totalLoad + " W";

return data.totalLoad;

}

/* ---------------- GRAPH ---------------- */

const ctx = document.getElementById('powerChart');

const powerChart = new Chart(ctx, {
type:'line',
data:{
labels:[],
datasets:[{
label:'Power Load',
data:[],
borderColor:'blue',
fill:false
}]
}
});

async function updateChart(){

const load = await fetchLoad();

const time = new Date().toLocaleTimeString();

powerChart.data.labels.push(time);
powerChart.data.datasets[0].data.push(load);

powerChart.update();

}

setInterval(updateChart,3000);


/* ---------------- DEVICE LIST ---------------- */

async function loadDevices(){

const response = await fetch("http://localhost:8080/devices/all");

const devices = await response.json();

let html = "";

devices.forEach(device => {

html += `
<p>
<b>${device.name}</b> - ${device.power}W - 
${device.status ? "ON" : "OFF"}
<button onclick="toggleDevice(${device.id})">Toggle</button>
</p>
`;

});

document.getElementById("deviceList").innerHTML = html;

}


/* ---------------- TOGGLE DEVICE ---------------- */

async function toggleDevice(id){

await fetch("http://localhost:8080/devices/toggle/" + id,{
method:"PUT"
});

loadDevices();

}


/* ---------------- INITIAL LOAD ---------------- */

loadDevices();
